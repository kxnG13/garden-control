    <script>
        const TOPIC_PREFIX = "myunique_garden_99/zone/"; // ต้องตรงกับใน ESP32
        const client = new Paho.MQTT.Client("broker.hivemq.com", 8000, "web_" + Math.random());

        client.connect({ onSuccess: () => {
            document.getElementById('status').innerHTML = "สถานะ: <span style='color:green'>ออนไลน์ ✅</span>";
        }});

        function send(zone, state) {
            let message = new Paho.MQTT.Message(state.toString());
            message.destinationName = TOPIC_PREFIX + zone;
            client.send(message);
        }

        // สร้างปุ่มแบบอัตโนมัติ 1-6
        const zones = ["หน้าบ้าน", "สวนกล้วยไม้", "แปลงผัก 1", "แปลงผัก 2", "ไม้ผล", "รั้วกินได้"];
        let html = "";
        zones.forEach((name, i) => {
            html += `<div class="btn-group">
                <span class="label">${name}</span>
                <div>
                    <button class="btn-on" onclick="send(${i+1}, 1)">เปิด</button>
                    <button class="btn-off" onclick="send(${i+1}, 0)">ปิด</button>
                </div>
            </div>`;
        });
        document.getElementById('controls').innerHTML = html;
    </script>
